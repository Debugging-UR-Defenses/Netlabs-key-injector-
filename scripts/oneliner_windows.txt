# WINDOWS INVENTORY - Run in CMD or PowerShell
# For CMD, use the first version. For PowerShell, use the second.

# === CMD VERSION (copy this single block) ===
@echo off & echo === %COMPUTERNAME% === & echo IP: & ipconfig | findstr /i "IPv4" & echo MAC: & getmac /fo list | findstr "Physical" & echo CPU: & wmic cpu get name /value | findstr "Name" & echo RAM: & wmic memorychip get capacity /value | findstr "Capacity" & echo DISK: & wmic diskdrive get size,model /value | findstr "Model Size" & echo OS: & wmic os get caption,version /value | findstr "Caption Version" & echo --- SOFTWARE --- & echo IE: & reg query "HKLM\Software\Microsoft\Internet Explorer" /v svcVersion 2>nul || reg query "HKLM\Software\Microsoft\Internet Explorer" /v Version & echo .NET: & reg query "HKLM\SOFTWARE\Microsoft\NET Framework Setup\NDP\v4\Full" /v Version 2>nul & echo PowerShell: & powershell -command "$PSVersionTable.PSVersion.ToString()" 2>nul & echo SMB: & wmic /namespace:\\root\Microsoft\Windows\SMB class MSFT_SmbServerConfiguration get EnableSMB1Protocol,EnableSMB2Protocol 2>nul & echo RDP: & reg query "HKLM\SYSTEM\CurrentControlSet\Control\Terminal Server" /v fDenyTSConnections & echo --- DRIVERS --- & driverquery /fo list | findstr "Module" | head -10 & echo === END ===

# === POWERSHELL VERSION (more reliable) ===
$h=$env:COMPUTERNAME;$ip=(Get-NetIPAddress -AddressFamily IPv4|?{$_.IPAddress -notlike "127.*"}|Select -First 1).IPAddress;$mac=(Get-NetAdapter|Select -First 1).MacAddress;$cpu=(Get-WmiObject Win32_Processor).Name;$ram=[math]::Round((Get-WmiObject Win32_ComputerSystem).TotalPhysicalMemory/1GB,1);$disk=[math]::Round((Get-WmiObject Win32_DiskDrive|Select -First 1).Size/1GB,0);$os=(Get-WmiObject Win32_OperatingSystem).Caption;$ver=(Get-WmiObject Win32_OperatingSystem).Version;Write-Host "=== $h ===";Write-Host "IP,$ip";Write-Host "MAC,$mac";Write-Host "CPU,$cpu";Write-Host "RAM,$ram GB";Write-Host "DISK,$disk GB";Write-Host "OS,$os";Write-Host "VER,$ver";Write-Host "--- SOFTWARE ---";Write-Host "PS,$($PSVersionTable.PSVersion)";Write-Host ".NET,$([System.Runtime.InteropServices.RuntimeInformation]::FrameworkDescription 2>$null)";Get-WmiObject Win32_Product 2>$null|Select -First 8|%{Write-Host "APP,$($_.Name),$($_.Version)"};Write-Host "--- DRIVERS ---";Get-WmiObject Win32_PnPSignedDriver|Select -First 10|%{Write-Host "DRV,$($_.DeviceName)"};Write-Host "=== END ==="
